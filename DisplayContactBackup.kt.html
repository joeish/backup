package com.backup.myapplication

import android.content.Intent
import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.ListView
import android.widget.ArrayAdapter
import android.widget.Button
import android.util.Log


class DisplayContactsActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_display_contact_backup)

        val contactsArray = intent.getSerializableExtra("contacts") as Array<Contact>
        val contacts = contactsArray.toList()


        val listView: ListView = findViewById(R.id.contactsListView)
        val adapter = ArrayAdapter(this, android.R.layout.simple_list_item_1, contacts.map { it.firstName })
        listView.adapter = adapter

        // Set the OnItemClickListener here
        listView.setOnItemClickListener { _, _, position, _ ->
            val selectedContact = contacts[position]
            val intent = Intent(this, ContactDetailsActivity::class.java)
            intent.putExtra("contact", selectedContact)
            startActivity(intent)
        }


    }
}
