package com.backup.myapplication

import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.TextView


class ContactDetailsActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_contact_details)

        val contact = intent.getSerializableExtra("contact") as Contact

        // Populate the UI elements with the contact details
        findViewById<TextView>(R.id.firstNameTextView).text = "First Name: ${contact.firstName}"
        findViewById<TextView>(R.id.middleNameTextView).text = "Middle Name: ${contact.middleName ?: "N/A"}"
        findViewById<TextView>(R.id.lastNameTextView).text = "Last Name: ${contact.lastName ?: "N/A"}"
        findViewById<TextView>(R.id.phoneNumbersTextView).text = "Phone Numbers: ${contact.phoneNumbers.joinToString(", ")}"
        findViewById<TextView>(R.id.emailsTextView).text = "Emails: ${contact.emails.joinToString(", ")}"
        findViewById<TextView>(R.id.companyTextView).text = "Company: ${contact.company ?: "N/A"}"
        findViewById<TextView>(R.id.departmentTextView).text = "Department: ${contact.department ?: "N/A"}"
        findViewById<TextView>(R.id.titleTextView).text = "Title: ${contact.title ?: "N/A"}"

        // For addresses, you might want to format them in a more readable way
        val addresses = contact.addresses.map { address ->
            listOf(
                address.street,
                address.city,
                address.region,
                address.postalCode,
                address.country
            ).filterNotNull().joinToString(", ")
        }
        findViewById<TextView>(R.id.addressesTextView).text = "Addresses: ${addresses.joinToString("\n")}"

        // For significant dates
        val dates = contact.significantDates.map { event ->
            "${event.date} (${event.description ?: "N/A"})"
        }
        findViewById<TextView>(R.id.significantDatesTextView).text = "Significant Dates: ${dates.joinToString("\n")}"

        findViewById<TextView>(R.id.websitesTextView).text = "Websites: ${contact.websites.joinToString(", ")}"
    }
}
